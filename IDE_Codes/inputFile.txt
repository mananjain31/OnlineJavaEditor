#include<iostream>
using namespace std;
class A
{	
	int x;
	public:
	friend istream& operator >> (istream& inp, A& a);
	friend ostream& operator << (ostream& outp, A& a);
};

istream& operator >> (istream& inp, A& a)
{
	inp >> a.x;
	return inp;
}

ostream& operator << (ostream& outp, A& a)
{
	outp << a.x;
	return outp;
}


int main()
{
A one, two;
cin >> one >> two ;
cout << "one is : " << one << endl;
cout << "two is : " << two << endl;
return 0;
}
